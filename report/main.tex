\section{Introduction}

We introduce basic graph theory concepts and briefly overview the results used in the project experiments.

\subsection{Signal processing on graphs}



We consider a weighted undirected graph $ \G = (\V, \E, \mathcal{W}) $,
where $ \V \subset \R^N $ is the set of vertices, $ \E \subset \R^M $ is the set of edges, and $ \mathcal{W}
: \V \times \mathcal{V} \to \R$ is a weight function. The weight function can be
represented with a $ N \times N $ matrix $ W $ such that 
$W_{i,j} = 
\begin{cases} 
    W(v_i, v_j), & \text{if } (v_i, v_j) \in \E \\ 
    0, & \text{otherwise} 
\end{cases} 
\quad \text{for all } i,j = 1, \dots, |\V|
$,
and for our needs, we assume $ W $ to be symmetric, that is $ W_{i,j} =  W_{j,i} $.

In the study of signal processing on graph, we model the idea of sending a signal to a node as
assigning a value to a vertex with a function $ s : \V \to \R$, whose values can be represented as a vector
$ s = s(\V) = [s_1, \dots, s_N] \in \R^N $ where each entry $ s_i \defeq s(v_i)$ represents
the signal sent over a node $ v_i \in \V$. We also keep track of the weight of each vertex
with a function $ d(i) \defeq \sum_{j=1}^N W_{i,j} $, that we represent with the matrix $D =
\text{diag}(d(1), \dots, d(N))$.

In this setting, we introduce the graph Laplacian $\L$ defined as 
$ \L = D - W$, where $D $ is the diagonal degree matrix with entries  $D_{ii} = d(i)$. By
construction, $\L^T = \mathcal{L}$, thus the Lanczos method can be safely applied to our 
problem.



\begin{remark}
  The Laplacian $\L$ is symmetric and semi-definite (it is diagonally dominant), hence by the
  spectral theorem it admits the decomposition
  \begin{equation*}
    \L = U \Lambda U^{*},
  \end{equation*}
  where $ U = \left[ u_0,\dots,u_{N-1} \right] \in O(N)$ is called Fourier basis, and $ \Lambda =
  \text{diag}\left(\lambda_0, \dots, \lambda_{N-1}\right)$ is the matrix of eigenvalues of $ \L $.
  Without loss of generality we can assume $ 0 =\lambda_0 \leq \lambda_1  \dots \leq \lambda_{N-1} $
  and the vectors in $ U $ to be in the same order that the eigenvalues.
  \end{remark}

\begin{definition}[Graph signal]
  A graph signal is a continuous function $ g : \R^+ \to \R $. For our needs, we will assume $ g $ to be
  defined on the spectrum of $ \L $.
\end{definition}
Diagonalising the Laplacian would give an easy way to compute the function $ g(\L) $ by
evaluating it on the eigenvalues of $ \L $, which means computing $ g(\L) = U g(\Lambda) U^{*} $.
In matrix notation, applying a filter to a graph $ \G $ corresponds to the operation
\begin{equation*}
  s^\prime \defeq g(\L)s = U g(\L) U^{*}s.
\end{equation*}
\begin{remark}[Computational cost]
Computing the Fourier basis for $ \L $ is computationally expensive for large graphs, because we
diagonalizing the Laplacian with a general method. In Section \ref{ssec:cost} we'll see
that Lanczos method for $ g(\L) $ offers an efficient alternative for computing $ g(\L) $.
\end{remark}

In this work, we will restrict ourself to two graph models: Erdős-Rényi and Sensor graphs. We shall
define them as the current implementation in Library \cite{pygsp} 
\begin{definition}[Erdős-Rényi graph]
  \label{def:ER}
  We define the Erdős-Rényi graph $ \G_{N,p} $ with $ N $ nodes, and probability $ p $, as the graph
  constructed by randomly connecting two nodes independently from any other edge, with probability $ p $.
\end{definition}
\begin{definition}[Sensor graph]
  \label{def:Sensor}
  A sensor graph $ \G_N $ is the graph constructed by picking $ N $ points on the square $ \left[
  0,1 \right] \times \left[ 0,1 \right] $ and connecting each point with edges using $ k $ nearest
  neighbors.
\end{definition}

\begin{proposition}
  The Laplacian matrix $ \L \in \R^{N\times N} $ satisfies the following properties:
  \begin{enumerate}
    \item $ \L $ is symmetric and positive semi-definite.
    \item The smallest eigenvalue of $ \L $ is $ 0 $ with corresponding eigenvector the constant one
      vector $ \mathbf{1} $.
    \item $ \L $ has $ N $ non-negative real-valued eigenvalues $ 0 = \lambda_1 \le \lambda_2 \le
      \ldots \le \lambda_n$.
  \end{enumerate}
\end{proposition}

\subsection{The Lanczos method}

\begin{definition}[Krylov subspace]
 Given a matrix $ A \in \R^{N \times N} $ and a vector $ b \in R^N $, the Krylov subspace of order
 $j$ is defined as the set $ \mathcal{K}_j (A,b) = \text{Span} \left(b, Ab, A^2b, \dots,
A^{j-1}b\right)  $. We represent the basis of this subspace with a matrix $ V_M = \left[ v_1,\dots,v_M
\right] \in \R^{N \times M}$.
\end{definition}
From the Arnoldi relation
\begin{equation*}
  AV_j = V_jH_j + h_{j+1,j}v_{j+1}e_j^{*}, \\ \hspace{20pt}
  H_j = V_j^{*}AV_j =
  \begin{bmatrix}
    h_{11} & \dots & \dots & h_{1,j} \\
    h_{21} & h_{22} & & \vdots \\
           & \ddots & \ddots & \vdots \\
           & & h_{j,j-1} & h_{j,j}.
  \end{bmatrix},
\end{equation*}
we can derive the Lanczos method for a symmetric, positively defined matrix $ A $. Setting
\begin{equation*}
\alpha_j \defeq h_{j,j},\hspace{20pt}\beta_j \defeq h_{j+1,j}.
\end{equation*}
From our hypothesis on $ A $, it follows that the matrix $ H_j $, which is both upper and lower Hessenberg 
matrix, is also symmetric; in other words, it is tridiagonal—thus we rename it $ T_j $. Hence
the Arnoldi relation takes the form:

\begin{equation*}
  AV_j = V_j \alpha_j  + \beta_j v_{j+1}e_j^{T}, \\ \hspace{20pt}
  T_j = V_j^{\top} A V_j = \begin{bmatrix}
    \alpha_1 & \beta_1 \\
    \beta_1 & \ddots & \ddots \\
      & \ddots & \ddots & \beta_{j-1} \\
      & & \beta_{j-1} & \alpha_j
  \end{bmatrix}.
\end{equation*}.

\begin{proposition}
 Consider the filter $ g : [0, \lambda_{\text{max}}] \to \R$ and a signal vector $ s \in \R^N $.
 Then the Lanczos method approximates $ g(\L)s $ with: \footnote{This results holds in general for any function $f :\Omega \to \C$, where $ \Omega \subset \Lambda(A) $.}
\begin{equation}
 g(\L)s \approx  V_M \norm{s}_2 g(T_M) e_1 \eqdef g_M. \label{eq:projection}
\end{equation}
\end{proposition}
Numerically, we compute the eigendecomposition $ (\Lambda,U) $ from $ T $, and then project $ g(T_M) = Ug(\Lambda)U^{\top} $ using the equation above.


\subsection{Algorithm}
\label{ssec:algo}

In floating point arithmetic the basis $ V $ suffers from loss of orthogonality, and orthogonalizing
the incoming vector $ w $ only against $ v_j,v_{j-1} $ becomes insufficient: it comes the need to
reorthogonalize the new vector $ w $ against all the previous basis vectors. To do so, we implement Algorithm \ref{alg:lanczos}, 
following \cite{susnjara2015} and \cite{demmel1997} with these criteria: full reorthogonalization is triggered if loss of
orthogonality is detected after calling Lanczos without orthogonalization, that is when $ V_M $ is
weakly orthogonal, that is $ \|V_M^{\top} V_M
- I \|_2 > \varepsilon_{\text{ORTHO}} $ for a chosen threshold $ \varepsilon_{\text{ORTHO}} = 10^{-10} $ \todo{Check threshold}. When the algorithm is called
passing a given threshold $ \varepsilon $, it computes the residual $
\|r_j \|_2  = \| g_j - g_{j-3}\|$, which computes $ g $ using Equation \ref{eq:projection},
recalling that $  g_M \defeq \|b \|_2V_Mg(T_M)e_1
$, where $ g(T_M) $ is computed by diagonalizing $ T_M $: because $ T_M $ is symmetric, banded tridiagonal, 
computing its eigendecomposition is stable and efficient; indeed the structure of $ T_M $ allows us
to use a divide and conqueer algorithm, and we'll use \texttt{eigh\_banded} routine fron Scipy.

\begin{remark}
  From now on we'll sometimes refer to Lanczos with double/full orthogonalization as full Lanczos and as
  Arnoldi with double orthogonalization as Arnoldi.
\end{remark}


\begin{algorithm}[ht!]
  \caption{Modified Lanczos method in floating point arithmetic.}
  \label{alg:lanczos}
  \begin{algorithmic}
    \State \textbf{Input:} Symmetric matrix $ A \in \R^{N\times N}$, vector $ b \in R^N $, $ M \in
    \N $, and $ g $ function.
    \State \textbf{Output: } Basis matrix $ V_l = \left[ v_1,\dots,v_l \right] \in R^{N \times l} $ for $ K_l(A,b) $, vectors $
    \alpha =\left\{ \alpha_1,\dots,\alpha_l \right\} \in \R^l, \beta=\left[ \beta_1,\dots,\beta_{l-1} \right] \in \R^{l-1} $, where $ l \in
    \{ 1,\dots,M \}$ is the stopping iteration index.
    \State Initialize $ V \leftarrow \mathbf{0}^{N \times M} $ , $ \alpha \leftarrow \mathbf{0}^N $,
    and $ \beta \leftarrow \mathbf{0}^{N-1}$

    \For{$ j = 1, \dots, M-1 $}
    \State $ w \leftarrow Av_j  $
    \State $ \alpha_j \leftarrow v_j^{\top} w $

    \State $\left\{
    \begin{array}{ll}
        w \leftarrow w - \alpha_j v_j&  \text{No reorthogonalization} \\
        \text{\textbf{if} } j>0 \text{ \textbf{then} } w \leftarrow w - \beta_j v_{j-1}\text{ \textbf{end
    if}} & \\ \\
        w = w - \sum_{i=1}^j (w^{\top}v_i)v_i & \text{Full reorthogonalization} \\
        w = w - \sum_{i=1}^j (w^{\top}v_i)v_i 
    \end{array}
    \right.$
    \If{$ j < M $}
    \State $ \beta_j = \|w\|_2 $
    \If{$ \beta_j = 0 $}
      \State 
      \Return $ V_j, \alpha_{1:j}, \beta_{1:j-1} $
      \Comment{--- Breakdown ---}
    \Else
      \If{$ \| g_j - g_{j-3}\|_2  < \varepsilon$}
      \State 
      \Return $ V_j, \alpha, \beta $
      \Comment{--- Breakdown ---}
      \EndIf
      \State $ v_{j+1} = \frac{w}{\beta_j} $
    \EndIf
    \EndIf
    \EndFor
    \State \Return $ V_M, \alpha_{1:M}, \beta_{1:M-1} $.
  \end{algorithmic}
  
\end{algorithm}

\begin{algorithm}[ht!]
  \caption{Approximation of $ g(\L)s $ using Lanczos.}
  \label{alg:g_evaluation}
  \begin{algorithmic}
    \State \textbf{Input:} Function $ g $, Matrix $ T $, signal $ s $.
    \State $ \lambda, U \leftarrow $ diagonalize($ T $)  \Comment{Here $ \lambda $ is the vector of
    eigenvalues.}
    \State $ u_1 \leftarrow  (U^{\top})_{:,1}$
    \State $y \leftarrow \|s\| U (\text{Diag}(g(\lambda)) U^{\top})e_1 = \|s\| U (g(\lambda) \odot u_1)$
    \State \Comment{Here $ u \odot v $ denotes the element-wise product between
  two vectors $ u $ and $ v $.}
    \State \Return $ Vy $
  \end{algorithmic}
\end{algorithm}

\begin{algorithm}[ht!]
  \caption{Evaluation of $ g(\L)s $ using Fourier basis.}
  \label{alg:g_evaluation_fourier}
  \begin{algorithmic}
    \State \textbf{Input:} Function $ g $, Graph $ \G $, signal $ s $.
    \State $ \Lambda,U \leftarrow \text{Fourier\_basis}( \G )$
    \State \Return $ U g(\Lambda) U^{\top}s $
  \end{algorithmic}
  
\end{algorithm}


\subsubsection{Computational cost}
\label{ssec:cost}

Here we overview computational cost of the algorithm that we tested, for an input matrix $ A $,
where $ \text{nnz}(A) $ is the number of non-zero elements of $ A $:

\begin{itemize}
  \item \textbf{Lanczos:} Computes $ V $  with partial orthogonalization, with $ O(M \cdot N
  + M \cdot \text{nnz}(A)) $ for a sparse matrix \footnote{note that the product $ Av_j $ dominates the cost,
  and that $ \text{nnz}(A) = N^2 $ when $ A $ is dense.}. When we need to evaluate function $ g $ at
  each iteraiton, such as in Section \ref{ssec:ex23}, we must consider also the $ O(N^2) $
  operations to compute $ g(\L) $ with divide and conqueer, thus the total cost in this case becomes $ O(M \cdot N^2) $.
\item \textbf{Lanczos algorithm with double orthogonalization:}, computing $ w = w -
  \sum_{i=1}^j v_i(v_i^{\top}w) $ can cost up
  to $ O(M\cdot N) $, thus the total cost becomes $ O(M^2N +  M\cdot \text{nnz}(A)) $, and when we
  also evaluate $ g $ it becomes $ O(M^2N + MN^2) $.
\item \textbf{Arnoldi with double orthogonalization}: For the orthogonalization it takes $ O(M^2N)
  $, plus the matrix vector product at each step with $ A $, hence again $ O(M^2N + M \cdot
  \text{nnz(A)}) $. Cost is higher when evaluating $ g $, due to the fact that matrix $ H $ is not
  perfectly symmetric and tridiagonal (it is up to a small epsilon). Thus we must use standard QR
  for computing the eigendecomposition, wich adds up to a brutal $ O(M \cdot N^3)$.
 \end{itemize}

We observe that in Lanczos computing $ g $ it not necessary a drawback when $ A $ is dense or when few
iterations are performed.
It should also be noted that these algorithms suffer from a significant $ O(M \cdot N) $ memory cost for storing $ V_M $.

\subsubsection{Comparing Lanczos with Arnoldi}

Here we do a short numerical comparison of Lanczos and Arnoldi with double orthogonalization, which
should reflect the cost analysis from Section \ref{ssec:cost}. Algorithm
\ref{alg:arnoldi_double_ortho} reports our implementation of Arnoldi.

\begin{algorithm}[ht!]
  \caption{Arnoldi with double orthogonalization.}
  \label{alg:arnoldi_double_ortho}
  \begin{algorithmic}[1]
    \State \textbf{Input:} Matrix $ A $, starting vector $ s $, maximum dimension $ M $, function $
    g $, tolerance $ \varepsilon $.
    \State $ v_1 \leftarrow s / \|s\|_2 $
    \State Initialize $ V \leftarrow \mathbf{0}^{N \times (M+1)} $ and $ H \leftarrow \mathbf{0}^{(M+1) \times M} $
    \State $ V_{:, 1} \leftarrow v_1 $
    \For{$ j = 1 \dots M $}
        \State $ w_1 \leftarrow A v_j $
        \State \Comment{First orthogonalization}
        \State $ h_1 \leftarrow V_j^\top w_1 $
        \State $ w_1 \leftarrow w_1 - V_j \cdot h_1 $
        \State $ v_{j+1} \leftarrow w_1 / \|w_1\|_2 $ 
        
        \State \Comment{Second orthogonalization pass}
        \State $ h_2 \leftarrow V_j^\top v_{j+1} $
        \State $ w_2 \leftarrow v_{j+1} - V_j \cdot h_2 $
        \State $ H_{1:j, j} \leftarrow h_1 + h_2 $
        \State $ H_{j+1, j} \leftarrow \|w_1\|_2 \|w_2\|_2 $\\
        
        \If{$ H_{j+1, j} = 0 $}
            \State \textbf{break} \Comment{Arnoldi breakdown}
        \EndIf
        
        \State \Comment{Convergence check}
        \If{$ j > 3 $ \textbf{and} $ j < M $}
            \If{$ \|g_j - g_{j-3}\|_2 < \varepsilon $}
                \State \Return $ V_j, H_{1:j, 1:j}, j $
            \EndIf
        \EndIf
        
        \State $ v_{j+1} \leftarrow w_2 / \|w_2|_2 $
        \State $ V_{:, j+1} \leftarrow v_{j+1} $
    \EndFor
    
    \State Set elements of $ H $ smaller than machine epsilon to $ 0 $
    \State \Return $ V_{:, 1:j}, H_{1:j, 1:j}, j $
  \end{algorithmic}
\end{algorithm}

\subsection{Performance comparison for Erdős-Rényi graphs}
\label{ssec:ex23}

sparsity, respectively. In fact, when $ N $ increases, we get expect higher computational cost due

$ becomes more dense, requiring more floating point operations for the algorithm and so increasing
runtime. This patter can be seen from Figure \ref{fig:spy}, which plots some matrices we used.
We choose $ \varepsilon_{\text{ORTHO}} = 10^{-5} $, as we experimentally observed that smaller thresholds rarely triggered full
orthogonalization. As described in Algorithm \ref{alg:lanczos} our stopping criteria happens when $ \|g_j - g_{j-3}\|_2 < \epsilon $, at the step $
j$. \footnote{Here
index goes backwards because matrix $ T_l $ would not be defined for $ l>j $ at iteration $ j $!}

\section{Conclusions \& future work}

We did't address the high memory cost of $ O(N\cdot M) $ of Lanczos, this could be avoided by
using a two-step implementation, that stores only three vectors and reconstructs the basis when
needed. Further, because full orthogonalization was widely used in the second experiment, we could
implement selective reorthogonalization as described in \cite{demmel1997}, but both methods fall
behind the objective of this course. Another improvement, which we already discussed, would be to
implement a working \texttt{WarpedTranslates} class in Library \cite{pygsp}.
